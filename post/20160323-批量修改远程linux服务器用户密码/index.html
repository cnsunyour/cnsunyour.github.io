批量修改远程 Linux 服务器用户密码

Posted by [不一般的凡] on Wed, Mar 23, 2016

首先是一个调用脚本，循环读取文本文件

文本文件格式为：ip loginuser loginpass passuser newpass

 1#!/bin/bash
 2if [ "$1" = "" ] || [ "$1" = "-h" ] || [ "$1" = "--help" ]
 3then
 4    echo "Usage: chpasswd.sh  path/iplist "
 5    exit
 6fi
 7cat $1 | while read line
 8do
 9    [ -z "$line" ] && continue
10    expect ~/expect/chpasswd.expect $line;
11done
12echo -e "\n  well done\n"

以下是 expect 主程序

 1#!/usr/bin/expect
 2
 3#登录的用户名
 4set loginuser [lindex $argv 1]
 5#密码
 6set loginpass  [lindex $argv 2]
 7
 8#要修改的用户名
 9set passuser [lindex $argv 3]
10#要修改成的新密码
11set newpass [lindex $argv 4]
12
13set ipaddr [lindex $argv 0]
14set timeout 300
15set cmd_prompt "]#|~]?"
16
17#---------------------------------------------------通过 ssh 登录
18spawn ssh $loginuser@$ipaddr
19set timeout 300
20expect {
21    -re "Are you sure you want to continue connecting (yes/no)?" {
22        send "yes\r"
23    } -re "assword:" {
24        send "$loginpass\r"
25    } -re "Permission denied, please try again." {
26        exit
27    } -re "Connection refused" {
28        exit
29    } -re $cmd_prompt {
30        send "\r"
31    } timeout {
32        exit
33    } eof {
34        exit
35    }
36}
37
38#-------------------------------------------修改密码
39send "sudo passwd $passuser \r";
40expect {
41    "password for $loginuser:" {
42        send "$loginpass\r"
43    }
44}
45
46expect {
47    "New password:" {
48        send "$newpass\r"
49    }
50    "新的 密码：" {
51        send "$newpass\r"
52    }
53    "passwd: Only root can specify a user name." {
54        exit
55    }
56}
57
58expect {
59    "Retype new password:" {
60        send "$newpass\r"
61    }
62    "重新输入新的 密码：" {
63        send "$newpass\r"
64    }
65}
66
67#---------------------------------------------退出
68expect -re $cmd_prompt
69exit
