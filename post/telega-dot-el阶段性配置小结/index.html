telega.el阶段性配置小结

Posted by [不一般的凡] on Sun, Jan 19, 2020

经过一段时间的调整，telega.el 的配置也基本固定了，都是一些细节上的小调整。另外自己加了个 advice，根据不同类型的 chatbuf 自动激活或关闭中文输入法，现在基本上做到进入 chatbuf 不需要切换输入法了。全部配置代码如下：

 1(use-package! telega
 2  :commands (telega)
 3  :defer t
 4  :bind ("C-M-S-s-t" . #'telega)
 5  :init
 6  (unless (display-graphic-p) (setq telega-use-images nil))
 7  :hook
 8  ('telega-chat-mode . #'yas-minor-mode-on)
 9  ('telega-chat-mode . (lambda ()
10                         (set-company-backend! 'telega-chat-mode
11                           (append '(telega-company-emoji
12                                     telega-company-username
13                                     telega-company-hashtag)
14                                   (when (telega-chat-bot-p telega-chatbuf--chat)
15                                     '(telega-company-botcmd))))))
16  ('telega-chat-pre-message . #'telega-msg-ignore-blocked-sender)
17  :config
18  (defadvice! +toggle-input-method--telega-chat-mode-a (chat)
19    "在 telega-chat-mode 里根据 chat 名称切换输入法，如果名称包含
20中文，则激活中文输入法，否则关闭中文输入法"
21    :after #'telega-chat--pop-to-buffer
22    (let ((input-method "pyim")
23          (title (telega-chat-title chat))
24          (cn-list (list "#archlinux-cn"
25                         "wikipedia-zh"
26                         "Jetbrains Agent"
27                         "SCP-079-CHAT"))
28          (en-list (list "telega.el")))
29      (cond ((member title cn-list) (activate-input-method input-method))
30            ((member title en-list) (activate-input-method nil))
31            ((string-match "\\cc" title) (activate-input-method input-method))
32            ((telega-chat-bot-p chat) (activate-input-method nil))
33            ((telega-chat-private-p chat) (activate-input-method input-method))
34            (t (activate-input-method nil)))))
35
36  (set-evil-initial-state! '(telega-root-mode telega-chat-mode) 'emacs)
37
38  (setq telega-proxies (list '(:server "127.0.0.1" :port 1086 :enable t
39                                       :type (:@type "proxyTypeSocks5")))
40        telega-chat-reply-prompt "<<< "
41        telega-chat-edit-prompt "+++ "
42        telega-chat-use-markdown-version nil
43        telega-animation-play-inline t
44        telega-emoji-use-images nil
45        telega-sticker-set-download t)
46  (pushnew! telega-known-inline-bots
47             "@vid" "@bing" "@wiki" "@imdb")
48
49  (set-popup-rule! (regexp-quote telega-root-buffer-name)
50    :side 'right :size 100 :quit t :modeline t)
51  (set-popup-rule! "^◀[[({<].*[\])}>]$"
52    :side 'right :size 100 :quit t :modeline t)
53
54  (telega-mode-line-mode 1)
55  (telega-url-shorten-mode 1)
56
57  (when (featurep! :completion ivy)
58    (load! "+ivy-telega"))
59
60  (after! all-the-icons
61    (add-to-list 'all-the-icons-mode-icon-alist
62                 '(telega-root-mode all-the-icons-fileicon "telegram"
63                                    :heigt 1.0
64                                    :v-adjust -0.2
65                                    :face all-the-icons-yellow))
66    (add-to-list 'all-the-icons-mode-icon-alist
67                 '(telega-chat-mode all-the-icons-fileicon "telegram"
68                                    :heigt 1.0
69                                    :v-adjust -0.2
70                                    :face all-the-icons-blue))))
